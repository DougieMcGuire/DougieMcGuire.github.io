    <script>
        let currentSlide = 1;
        showSlide(currentSlide);

        function changeSlide(n) {
            showSlide(currentSlide += n);
        }

        function showSlide(n) {
            const slides = document.querySelectorAll('.slide');
            if (n > slides.length) { currentSlide = 1; }
            if (n < 1) { currentSlide = slides.length; }
            slides.forEach((slide) => slide.style.display = 'none');
            slides[currentSlide - 1].style.display = 'block';
        }

        // Fetch and insert the bookmarklet code from blmkcode.txt
        fetch('blmkcode.txt')
            .then(response => response.text())
            .then(bookmarkletCode => {
                // Set up the drag link with the fetched code
                document.getElementById('dragLink').href = `javascript:(function(){${bookmarkletCode}})();`;
            })
            .catch(error => console.error('Error loading bookmarklet code:', error));
    </script>
